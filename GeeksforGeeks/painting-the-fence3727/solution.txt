// Problem: Painting the Fence
// Difficulty: Unknown

class Solution {
    int countWays(int n, int k) {
        if (n == 1) return k;
        int same = 0;
        int diff = k;
        for (int i = 2; i <= n; i++) {
            int newSame = diff;
            int newDiff = (same + diff) * (k - 1);
            same = newSame;
            diff = newDiff;
        }
        return same + diff;
    }
}
