// Problem: Buy Stock with Transaction Fee
// Difficulty: Unknown
// Source: GeeksforGeeks
// Language: Java (21)

class Solution {
    public int maxProfit(int arr[], int k) {
        int n = arr.length;
        if (n <= 1) return 0;
        int hold = -arr[0];
        int free = 0;
        for (int i = 1; i < n; i++) {
            int nextFree = Math.max(free, hold + arr[i] - k);
            int nextHold = Math.max(hold, free - arr[i]);
            free = nextFree;
            hold = nextHold;
        }
        return free;
    }
}
